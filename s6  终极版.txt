bool displayFeedbackActive = false;

const unsigned long feedbackDuration = 700;

/* ================= 辅助工具函数 ================= */

// 信号强度转换：将 -100~0 映射为 0~100 (数值越大信号越好)
int getSignalScore(int rssi) {
  if (rssi == 0) return 0;
  return constrain(rssi + 100, 0, 100); 
}

/* ================= 触摸检测 ================= */
bool isTouching() {
  pinMode(TOUCH_PIN, OUTPUT);       
  digitalWrite(TOUCH_PIN, LOW);    
  delayMicroseconds(1000);         
  pinMode(TOUCH_PIN, INPUT_PULLUP); 
  uint32_t count = 0;              
  while (digitalRead(TOUCH_PIN) == LOW && count < 2000) { 
    count++;                       
    asm volatile ("nop");          
  }
  return (count > 2);
}

OneButton button;

String getStatusJSON() {
  int rssi = (WiFi.status() == WL_CONNECTED) ? WiFi.RSSI() : 0;
  bool default_ap = preferences.getBool("default_ap", false);
  return "{\"mode\":" + String((int)randomMode) + 
         ",\"lock\":" + String(isEyeLocked ? 1 : 0) + 
         ",\"saved_ssid\":\"" + sta_ssid + "\"" +
         ",\"ip\":\"" + (isAPMode ? WiFi.softAPIP().toString() : WiFi.localIP().toString()) + "\"" +
         ",\"rssi\":" + String(getSignalScore(rssi)) + 
         ",\"default_ap\":" + String(default_ap ? "true" : "false") + "}";
}

/* ================= 屏幕显示控制 ================= */

// 开启 AP 热点模式
void startAP() {
  isAPMode = true;
  WiFi.mode(WIFI_AP);
  WiFi.softAP("桌面机器人");
  
  // 4秒倒计时界面
  for (int i = 4; i > 0; i--) {
    display.clearDisplay();
    u8g2Fonts.setFont(u8g2_font_wqy12_t_gb2312);
    u8g2Fonts.setCursor(0, 15); u8g2Fonts.print("热点模式");
    u8g2Fonts.setCursor(0, 31); u8g2Fonts.print("请连接WiFi：桌面机器人");
    u8g2Fonts.setCursor(0, 47); u8g2Fonts.print("ip地址：192.168.4.1");
    u8g2Fonts.setCursor(0, 62); u8g2Fonts.print(String(i) + "秒后关闭该界面");
    display.display();
    delay(1000);
  }
}

// 初始化 WiFi (STA 或 AP)
void initWiFi() {
  preferences.begin("wifi-cfg", false);
  sta_ssid = preferences.getString("ssid", "");
  sta_password = preferences.getString("password", "");
  bool default_ap = preferences.getBool("default_ap", false);

  if (default_ap || sta_ssid == "") {
    startAP();
  } else {
    WiFi.mode(WIFI_STA);
    WiFi.begin(sta_ssid.c_str(), sta_password.c_str());
    
    unsigned long startTime = millis();
    bool connected = false;
    
    // 5秒内尝试连接并实时更新屏幕状态
    while (millis() - startTime < 5000) {
      if (WiFi.status() == WL_CONNECTED) {
        connected = true;
        break;
      }
      
      display.clearDisplay();
      u8g2Fonts.setFont(u8g2_font_wqy12_t_gb2312);
      u8g2Fonts.setCursor(0, 15); u8g2Fonts.print("WiFi模式：正在连接");
      u8g2Fonts.setCursor(0, 31); u8g2Fonts.print(sta_ssid);
      u8g2Fonts.setCursor(0, 47); u8g2Fonts.print("ip地址：" + WiFi.localIP().toString());
      u8g2Fonts.setCursor(0, 62); u8g2Fonts.print("信号强度" + String(getSignalScore(WiFi.RSSI())) + "dBm");
      display.display();
      delay(200); 
    }
    
    if (connected) {
      isAPMode = false;
      // --- 成功连接后的 4 秒倒计时界面 (按照最新要求修改) ---
      for (int i = 4; i > 0; i--) {
        display.clearDisplay();
        u8g2Fonts.setFont(u8g2_font_wqy12_t_gb2312);
        u8g2Fonts.setCursor(0, 15); u8g2Fonts.print("WiFi模式：已连接");
        u8g2Fonts.setCursor(0, 31); u8g2Fonts.print(sta_ssid);
        u8g2Fonts.setCursor(0, 47); u8g2Fonts.print("ip地址：" + WiFi.localIP().toString());
        u8g2Fonts.setCursor(0, 62); 
        // 将信号强度和倒计时显示在最后一行
        u8g2Fonts.print("信号强度" + String(getSignalScore(WiFi.RSSI())) + "dBm;  " + String(i) + "s");
        display.display();
        delay(1000);
      }
    } else {
      // 5秒内未连上，倒计时后切回热点
      preferences.putBool("default_ap", true); 
      for (int i = 4; i > 0; i--) {
        display.clearDisplay();
        u8g2Fonts.setFont(u8g2_font_wqy12_t_gb2312);
        u8g2Fonts.setCursor(0, 15); u8g2Fonts.print("WiFi模式：连接失败");
        u8g2Fonts.setCursor(0, 31); u8g2Fonts.print("请重新配置WiFi...");
        u8g2Fonts.setCursor(0, 55); u8g2Fonts.print(String(i) + "秒后进入热点模式");
        display.display();
        delay(1000);
      }
      startAP();
    }
  }
  display.clearDisplay();
  display.display();
}

/* ================= 电机控制 ================= */
void motorWifi(byte c) {
  digitalWrite(STBY, HIGH);
  switch (c) {
    case 0: digitalWrite(LF, LOW); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, LOW); break;
    case 1: digitalWrite(LF, HIGH); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, HIGH); break;
    case 2: digitalWrite(LF, LOW); digitalWrite(LB, HIGH); digitalWrite(RF, HIGH); digitalWrite(RB, LOW); break;
    case 3: digitalWrite(LF, LOW); digitalWrite(LB, HIGH); digitalWrite(RF, LOW); digitalWrite(RB, HIGH); break;
    case 4: digitalWrite(LF, HIGH); digitalWrite(LB, LOW); digitalWrite(RF, HIGH); digitalWrite(RB, LOW); break;
  }
}

void handleMovement(byte dir) {
  motorWifi(dir);
  lastWebControlTime = millis();
  motorStopTimer = millis() + 10000;
  if (!isAutoSleeping) {
    backupMode = randomMode;
    randomMode = RANDOM_OFF;
    isAutoSleeping = true;
  }
  modeRestoreTimer = millis() + 15000;
}

void MOTOR(byte c, int t1, int t2, int Time) {
  for (int i = 0; i < Time; i++) {
    switch (c) {
      case 0: digitalWrite(LF, LOW); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, LOW); break;
      case 1: digitalWrite(LF, LOW); digitalWrite(LB, HIGH); digitalWrite(RF, LOW); digitalWrite(RB, HIGH); break;
      case 2: digitalWrite(LF, HIGH); digitalWrite(LB, LOW); digitalWrite(RF, HIGH); digitalWrite(RB, LOW); break;
      case 3: digitalWrite(LF, LOW); digitalWrite(LB, HIGH); digitalWrite(RF, HIGH); digitalWrite(RB, LOW); break;
      case 4: digitalWrite(LF, HIGH); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, HIGH); break;
      case 5: digitalWrite(LF, LOW); digitalWrite(LB, HIGH); digitalWrite(RF, LOW); digitalWrite(RB, LOW); break;
      case 6: digitalWrite(LF, LOW); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, HIGH); break;
      case 7: digitalWrite(LF, HIGH); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, LOW); break;
      case 8: digitalWrite(LF, LOW); digitalWrite(LB, LOW); digitalWrite(RF, HIGH); digitalWrite(RB, LOW); break;
    }
    delay(t1);
    digitalWrite(LF, LOW); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, LOW);
    delay(t2);
  }
}

/* ================= 核心交互逻辑 ================= */
void triggerModeSwitch() {
  if (randomMode == RANDOM_OFF) randomMode = RANDOM_SOFT;
  else if (randomMode == RANDOM_SOFT) randomMode = RANDOM_NORMAL;
  else randomMode = RANDOM_OFF;
  
  feedbackText = (randomMode == RANDOM_OFF) ? "睡眠模式" : (randomMode == RANDOM_SOFT ? "摆动模式" : "好奇模式");
  displayFeedbackActive = true;
  feedbackStartTime = millis();
}

void triggerLockExpression() {
  isEyeLocked = !isEyeLocked;
  feedbackText = isEyeLocked ? "表情已锁定" : "表情已解锁";
  displayFeedbackActive = true;
  feedbackStartTime = millis();
}

void setup() {
  Serial.begin(115200);
  pinMode(STBY, OUTPUT); digitalWrite(STBY, LOW);
  pinMode(LF, OUTPUT); pinMode(LB, OUTPUT); pinMode(RF, OUTPUT); pinMode(RB, OUTPUT);
  Wire.begin(OLED_SDA, OLED_SCL);
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  
  u8g2Fonts.begin(display);
  u8g2Fonts.setFontMode(1);                 
  u8g2Fonts.setFontDirection(0);            
  u8g2Fonts.setForegroundColor(SSD1306_WHITE); 
  
  initWiFi(); 

  roboEyes.begin(SCREEN_WIDTH, SCREEN_HEIGHT, 100);
  roboEyes.setAutoblinker(ON, 3, 2);
  roboEyes.setIdleMode(ON, 2, 2);
  roboEyes.setMood(moods[currentMoodIdx]);

  button.attachClick([](){ isEyeLocked=false; currentMoodIdx=(currentMoodIdx+1)%4; roboEyes.setMood(moods[currentMoodIdx]); lastMoodTime=millis(); });
  button.attachDoubleClick(triggerModeSwitch);
  button.attachLongPressStart(triggerLockExpression);

  dnsServer.start(53, "*", WiFi.softAPIP());
  
  server.on("/", handleRoot);
  server.on("/status", [](){ server.send(200, "application/json", getStatusJSON()); });
  server.on("/f", [](){ handleMovement(1); server.send(200, "application/json", getStatusJSON()); });
  server.on("/b", [](){ handleMovement(2); server.send(200, "application/json", getStatusJSON()); });
  server.on("/l", [](){ handleMovement(3); server.send(200, "application/json", getStatusJSON()); });
  server.on("/r", [](){ handleMovement(4); server.send(200, "application/json", getStatusJSON()); });
  server.on("/s", [](){ motorWifi(0); motorStopTimer=0; server.send(200, "application/json", getStatusJSON()); });
  server.on("/mode_off", [](){ randomMode=RANDOM_OFF; server.send(200, "application/json", getStatusJSON()); });
  server.on("/mode_soft", [](){ randomMode=RANDOM_SOFT; server.send(200, "application/json", getStatusJSON()); });
  server.on("/mode_normal", [](){ randomMode=RANDOM_NORMAL; server.send(200, "application/json", getStatusJSON()); });
  server.on("/expr_next", [](){ isEyeLocked=false; currentMoodIdx=(currentMoodIdx+1)%4; roboEyes.setMood(moods[currentMoodIdx]); server.send(200, "application/json", getStatusJSON()); });
  server.on("/expr_lock", [](){ triggerLockExpression(); server.send(200, "application/json", getStatusJSON()); });

  server.on("/toggle_mode", [](){
    bool current_ap = preferences.getBool("default_ap", false);
    preferences.putBool("default_ap", !current_ap);
    server.send(200, "text/plain", "已切换模式，正在重启");
    delay(1000); ESP.restart();
  });

  server.on("/set_wifi", [](){
    sta_ssid = server.arg("s");
    sta_password = server.arg("p");
    preferences.putString("ssid", sta_ssid);
    preferences.putString("password", sta_password);
    server.send(200, "text/plain", "OK");
  });

  server.onNotFound([]() {
    if (isAPMode) {
      server.sendHeader("Location", String("http://") + WiFi.softAPIP().toString(), true);
      server.send(302, "text/plain", "");
    } else {
      server.send(404, "text/plain", "Not Found");
    }
  });

  server.begin();
  digitalWrite(STBY, HIGH);
}

void loop() {
  dnsServer.processNextRequest();
  server.handleClient();
  if (motorStopTimer > 0 && millis() > motorStopTimer) { motorWifi(0); motorStopTimer = 0; }
  if (isAutoSleeping && millis() > modeRestoreTimer) { randomMode = backupMode; isAutoSleeping = false; }
  if (millis() - lastWebControlTime > 5000) { button.tick(isTouching()); }

  if (displayFeedbackActive) {
    if (millis() - feedbackStartTime < feedbackDuration) {
      display.clearDisplay();
      u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312); 
      int16_t w = u8g2Fonts.getUTF8Width(feedbackText.c_str());
      u8g2Fonts.setCursor((128 - w) / 2, 40); 
      u8g2Fonts.print(feedbackText);
      display.display();
    } else { 
      displayFeedbackActive = false; 
    }
  } else { 
    roboEyes.update(); 
  }

  if (!isEyeLocked && (millis() - lastMoodTime >= moodInterval)) {
    lastMoodTime = millis();
    currentMoodIdx = (currentMoodIdx + 1) % 4;
    roboEyes.setMood(moods[currentMoodIdx]);
  }

  static unsigned long lastMoveTick = 0;
  if (millis() - lastMoveTick > 40) {
    lastMoveTick = millis();
    if (randomMode == RANDOM_SOFT && random(120) == 1) MOTOR(random(9), 25, 60, 1);
    else if (randomMode == RANDOM_NORMAL && random(100) == 1) MOTOR(random(9), 15, 50, random(15));
  }
}
