bool isTouching() {
  pinMode(TOUCH_PIN, OUTPUT);
  digitalWrite(TOUCH_PIN, LOW);
  delayMicroseconds(1000);
  pinMode(TOUCH_PIN, INPUT_PULLUP);
  uint32_t count = 0;
  while (digitalRead(TOUCH_PIN) == LOW && count < 2000) { count++; asm volatile ("nop"); }
  return (count > 2);
}

void drawProgressBarBorder(float progress) {
  int remainingPixels = 384 * progress;
  int start = 384 - remainingPixels;
  int end = 384;
  if (start >= end) return;
  if (start < 128) { int s = start; int e = min(end, 128); display.drawFastHLine(s, 0, e - s, SSD1306_WHITE); }
  if (end > 128 && start < 192) { int s = max(start, 128) - 128; int e = min(end, 192) - 128; display.drawFastVLine(127, s, e - s, SSD1306_WHITE); }

String getStatusJSON() {
  int rssi = (WiFi.status() == WL_CONNECTED) ? WiFi.RSSI() : 0;
  bool default_ap = preferences.getBool("default_ap", false);
  int reportMode = (int)backupMode; // 反馈新模式，即使不执行

  String cdActive = (currentState == STATE_COUNTDOWN) ? "true" : "false";
  String cdRem = "";
  if (currentState == STATE_COUNTDOWN) {
    unsigned long elapsed = millis() - countdownStartTime;
    unsigned long rem = (countdownDurationMs > elapsed) ? (countdownDurationMs - elapsed) : 0;
    int rMin = (rem / 1000) / 60;
    int rSec = (rem / 1000) % 60;
    char buf[10]; sprintf(buf, "%02d:%02d", rMin, rSec);
    cdRem = String(buf);
  }

  return "{\"mode\":" + String(reportMode) +
  ",\"lock\":" + String((isEyeLocked || customTextLocked) ? 1 : 0) +
  ",\"saved_ssid\":\"" + sta_ssid + "\"" +
  ",\"ip\":\"" + (isAPMode ? WiFi.softAPIP().toString() : WiFi.localIP().toString()) + "\"" +
  ",\"rssi\":" + String(getSignalScore(rssi)) +
  ",\"default_ap\":" + String(default_ap ? "true" : "false") +
  ",\"cd_active\":" + cdActive +
  ",\"cd_rem\":\"" + cdRem + "\"}";
}

/* ================= 触控动作逻辑 ================= */
void triggerModeSwitch() {
  if (backupMode == RANDOM_OFF) backupMode = RANDOM_SOFT;
  else if (backupMode == RANDOM_SOFT) backupMode = RANDOM_NORMAL;
  else backupMode = RANDOM_OFF;
  
  if(currentState != STATE_COUNTDOWN) randomMode = backupMode;
  
  feedbackText = (backupMode == RANDOM_OFF) ? "睡眠模式" : (backupMode == RANDOM_SOFT ? "摆动模式" : "好奇模式");
  displayFeedbackActive = true; feedbackStartTime = millis();
}

void triggerLockExpression() {
  if (customTextActive) {
    customTextLocked = !customTextLocked;
    feedbackText = customTextLocked ? "文字已锁定" : "显示已取消";
  } else {
    isEyeLocked = !isEyeLocked;
    feedbackText = isEyeLocked ? "表情已锁定" : "表情已解锁";
  }
  displayFeedbackActive = true; feedbackStartTime = millis();
}

void handleSingleClick() {
  if (currentState == STATE_NORMAL) {
    if (customTextActive) { customTextActive = false; customTextLocked = false; }
    isEyeLocked = false; currentMoodIdx = (currentMoodIdx + 1) % 4;
    roboEyes.setMood(moods[currentMoodIdx]); lastMoodTime = millis();
  }
  else if (currentState == STATE_MENU_TIME) { menuIndex = (menuIndex + 1) % 11; }
  else if (currentState == STATE_MENU_CONFIRM) { confirmMenuSelection = (confirmMenuSelection == 0) ? 1 : 0; }
  else if (currentState == STATE_MENU_CUSTOM) {
    currentSelectedMinutes++;
    if (currentSelectedMinutes >= customMaxMinutes) currentSelectedMinutes = customBaseMinutes;
  }
  else if (currentState == STATE_COUNTDOWN) {
    long remSec = ((countdownDurationMs - (millis() - countdownStartTime)) / 1000);
    if (remSec <= 55 && remSec >= 10) return;
    if (showingFaceInCountdown) {
        currentMoodIdx = (currentMoodIdx + 1) % 4;
        roboEyes.setMood(moods[currentMoodIdx]);
        lastFaceShowTime = millis();
    } else {
        showingFaceInCountdown = true;
        lastFaceShowTime = millis();
    }
  }
}

void handleDoubleClick() {
  if (currentState == STATE_NORMAL || currentState == STATE_COUNTDOWN) { triggerModeSwitch(); }
  else if (currentState == STATE_MENU_TIME) { menuIndex = (menuIndex - 1 + 11) % 11; }
  else if (currentState == STATE_MENU_CUSTOM) {
    currentSelectedMinutes--;
    if (currentSelectedMinutes < customBaseMinutes) currentSelectedMinutes = customMaxMinutes - 1;
  }
}

void handleTripleClick() {
  if (currentState == STATE_NORMAL) { triggerLockExpression(); }
}

void handleLongPress() {
  if (currentState == STATE_NORMAL) { currentState = STATE_MENU_TIME; menuIndex = 0; }
  else if (currentState == STATE_MENU_TIME) {
    currentSelectedMinutes = presetTimes[menuIndex];
    if (currentSelectedMinutes >= 5) { currentState = STATE_MENU_CONFIRM; confirmMenuSelection = 0; }
    else { startCountdown(currentSelectedMinutes); }
  }
  else if (currentState == STATE_MENU_CONFIRM) {
    if (confirmMenuSelection == 0) { startCountdown(currentSelectedMinutes); }
    else {
      currentState = STATE_MENU_CUSTOM; customBaseMinutes = currentSelectedMinutes;
      customMaxMinutes = currentSelectedMinutes + 15;
      for(int i=0; i<10; i++) { if(presetTimes[i] == currentSelectedMinutes) { customMaxMinutes = presetTimes[i+1]; break; } }
    }
  }
  else if (currentState == STATE_MENU_CUSTOM) { startCountdown(currentSelectedMinutes); }
  else if (currentState == STATE_COUNTDOWN) {
    currentState = STATE_NORMAL; randomMode = backupMode;
    displayFeedbackActive = true; feedbackText = "倒计时解除"; feedbackStartTime = millis();
  }
}

void analyzeCustomText() {
  int totalWidth = u8g2Fonts.getUTF8Width(customTextContent.c_str());
  int totalChars = 0; int i = 0; int len = customTextContent.length();
  while (i < len) {
    char c = customTextContent[i];
    if ((c & 0x80) == 0) i += 1; else if ((c & 0xE0) == 0xC0) i += 2; else if ((c & 0xF0) == 0xE0) i += 3; else if ((c & 0xF8) == 0xF0) i += 4;
    totalChars++;
  }
  if (totalWidth <= 112 && totalChars <= 7) { textDisplayMode = 1; textLine1 = customTextContent; }
  else {
    textLine1 = ""; textLine2 = ""; int w1 = 0; i = 0; int charCount1 = 0;
    while (i < len) {
      char c = customTextContent[i]; int charLen = 1;
      if ((c & 0x80) == 0) charLen = 1; else if ((c & 0xE0) == 0xC0) charLen = 2; else if ((c & 0xF0) == 0xE0) charLen = 3; else if ((c & 0xF8) == 0xF0) charLen = 4;
      String singleChar = customTextContent.substring(i, i + charLen);
      int cw = u8g2Fonts.getUTF8Width(singleChar.c_str());
      if (charCount1 < 7 && w1 + cw <= 112) { textLine1 += singleChar; w1 += cw; charCount1++; }
      else { textLine2 += customTextContent.substring(i); break; }
      i += charLen;
    }
    textDisplayMode = (u8g2Fonts.getUTF8Width(textLine2.c_str()) <= 112 && (totalChars - charCount1) <= 7) ? 2 : 3;
    scrollX = 128;
  }
}

void initWiFi() {
  preferences.begin("wifi-cfg", false);
  sta_ssid = preferences.getString("ssid", ""); sta_password = preferences.getString("password", ""); ap_ssid = preferences.getString("ap_ssid", "桌面机器人");
  if (preferences.getBool("default_ap", false) || sta_ssid == "") { startAP(); }
  else {
    WiFi.mode(WIFI_STA); WiFi.begin(sta_ssid.c_str(), sta_password.c_str());
    unsigned long startTime = millis(); bool connected = false;
    while (millis() - startTime < 5000) {
      if (WiFi.status() == WL_CONNECTED) { connected = true; break; }
      display.clearDisplay(); u8g2Fonts.setFont(u8g2_font_wqy12_t_gb2312);
      u8g2Fonts.setCursor(0, 15); u8g2Fonts.print("WiFi模式：正在连接");
      u8g2Fonts.setCursor(0, 31); u8g2Fonts.print(sta_ssid);
      u8g2Fonts.setCursor(0, 47); u8g2Fonts.print("ip地址：" + WiFi.localIP().toString());
      u8g2Fonts.setCursor(0, 62); u8g2Fonts.print("信号强度：" + String(getSignalScore(WiFi.RSSI())) + "dBm");
      display.display(); delay(200);
    }
    if (connected) {
      isAPMode = false;
      for (int i = 3; i > 0; i--) {
        display.clearDisplay(); u8g2Fonts.setFont(u8g2_font_wqy12_t_gb2312);
        u8g2Fonts.setCursor(0, 15); u8g2Fonts.print("WiFi模式：已连接");
        u8g2Fonts.setCursor(0, 31); u8g2Fonts.print(sta_ssid);
        u8g2Fonts.setCursor(0, 47); u8g2Fonts.print("ip地址：" + WiFi.localIP().toString());
        u8g2Fonts.setCursor(0, 62); u8g2Fonts.print("信号强度：" + String(getSignalScore(WiFi.RSSI())) + "dBm，" + String(i) + "s");
        display.display(); delay(1000);
      }
    } else { preferences.putBool("default_ap", true); startAP(); }
  }
  display.clearDisplay(); display.display();
}

void motorWifi(byte c) {
  digitalWrite(STBY, HIGH);
  switch (c) {
    case 0: digitalWrite(LF, LOW); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, LOW); break;
    case 1: digitalWrite(LF, HIGH); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, HIGH); break;
    case 2: digitalWrite(LF, LOW); digitalWrite(LB, HIGH); digitalWrite(RF, HIGH); digitalWrite(RB, LOW); break;
    case 3: digitalWrite(LF, LOW); digitalWrite(LB, HIGH); digitalWrite(RF, LOW); digitalWrite(RB, HIGH); break;
    case 4: digitalWrite(LF, HIGH); digitalWrite(LB, LOW); digitalWrite(RF, HIGH); digitalWrite(RB, LOW); break;
  }
}

void MOTOR(byte c, int t1, int t2, int Time) {
  if (randomMode == RANDOM_OFF || currentState == STATE_COUNTDOWN) return;
  for (int i = 0; i < Time; i++) {
    switch (c) {
      case 0: digitalWrite(LF, LOW); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, LOW); break;
      case 1: digitalWrite(LF, LOW); digitalWrite(LB, HIGH); digitalWrite(RF, LOW); digitalWrite(RB, HIGH); break;
      case 2: digitalWrite(LF, HIGH); digitalWrite(LB, LOW); digitalWrite(RF, HIGH); digitalWrite(RB, LOW); break;
      case 3: digitalWrite(LF, LOW); digitalWrite(LB, HIGH); digitalWrite(RF, HIGH); digitalWrite(RB, LOW); break;
      case 4: digitalWrite(LF, HIGH); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, HIGH); break;
      case 5: digitalWrite(LF, LOW); digitalWrite(LB, HIGH); digitalWrite(RF, LOW); digitalWrite(RB, LOW); break;
      case 6: digitalWrite(LF, LOW); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, HIGH); break;
      case 7: digitalWrite(LF, HIGH); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, LOW); break;
      case 8: digitalWrite(LF, LOW); digitalWrite(LB, LOW); digitalWrite(RF, HIGH); digitalWrite(RB, LOW); break;
    }
    unsigned long start = millis(); while(millis() - start < t1) { server.handleClient(); yield(); }
    digitalWrite(LF, LOW); digitalWrite(LB, LOW); digitalWrite(RF, LOW); digitalWrite(RB, LOW);
    start = millis(); while(millis() - start < t2) { server.handleClient(); yield(); }
  }
}

void setup() {
  Serial.begin(115200);
  pinMode(STBY, OUTPUT); digitalWrite(STBY, LOW);
  pinMode(LF, OUTPUT); pinMode(LB, OUTPUT); pinMode(RF, OUTPUT); pinMode(RB, OUTPUT);
  Wire.begin(OLED_SDA, OLED_SCL);
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  u8g2Fonts.begin(display);
  u8g2Fonts.setFontMode(1); u8g2Fonts.setForegroundColor(SSD1306_WHITE);
  initWiFi();
  roboEyes.begin(SCREEN_WIDTH, SCREEN_HEIGHT, 100);
  roboEyes.setAutoblinker(true, 3, 2); 
  roboEyes.setIdleMode(true, 2, 2);
  roboEyes.setMood(moods[currentMoodIdx]);
  dnsServer.start(53, "*", WiFi.softAPIP());
  server.on("/f", []() { handleMovement(1); server.send(204); });
  server.on("/b", []() { handleMovement(2); server.send(204); });
  server.on("/l", []() { handleMovement(3); server.send(204); });
  server.on("/r", []() { handleMovement(4); server.send(204); });
  server.on("/s", []() { motorWifi(0); motorStopTimer=0; server.send(204); });
  server.on("/", handleRoot);
  server.on("/status", []() { server.send(200, "application/json", getStatusJSON()); });
  server.on("/mode_off", []() { backupMode=RANDOM_OFF; if(currentState!=STATE_COUNTDOWN) randomMode=RANDOM_OFF; server.send(200, "application/json", getStatusJSON()); });
  server.on("/mode_soft", []() { backupMode=RANDOM_SOFT; if(currentState!=STATE_COUNTDOWN) randomMode=RANDOM_SOFT; server.send(200, "application/json", getStatusJSON()); });
  server.on("/mode_normal", []() { backupMode=RANDOM_NORMAL; if(currentState!=STATE_COUNTDOWN) randomMode=RANDOM_NORMAL; server.send(200, "application/json", getStatusJSON()); });
  server.on("/expr_next", []() {
    if (customTextActive) { customTextActive = false; customTextLocked = false; }
    isEyeLocked=false; currentMoodIdx=(currentMoodIdx+1)%4; roboEyes.setMood(moods[currentMoodIdx]);
    server.send(200, "application/json", getStatusJSON());
  });
  server.on("/expr_lock", []() { triggerLockExpression(); server.send(200, "application/json", getStatusJSON()); });
  server.on("/toggle_mode", []() { preferences.putBool("default_ap", !preferences.getBool("default_ap", false)); server.send(200, "text/plain", "OK"); delay(500); ESP.restart(); });
  server.on("/set_wifi", []() { preferences.putString("ssid", server.arg("s")); preferences.putString("password", server.arg("p")); server.send(200, "text/plain", "OK"); });
  server.on("/set_ap_name", []() { preferences.putString("ap_ssid", server.arg("ap")); server.send(200, "text/plain", "OK"); });
  server.on("/start_timer", []() { int m = server.arg("m").toInt(); if(m>0) { startCountdown(m); server.send(200, "text/plain", "OK"); } });
  server.on("/stop_timer", []() { currentState = STATE_NORMAL; randomMode = backupMode; server.send(200, "text/plain", "OK"); });
  server.on("/display_text", []() {
    customTextContent = server.arg("msg");
    if (customTextContent == "") { customTextActive = false; customTextLocked = false; }
    else { customTextActive = true; customTextLocked = false; customTextStartTime = millis(); analyzeCustomText(); }
    server.send(200, "text/plain", "OK");
  });
  server.onNotFound([]() { if(isAPMode) { server.sendHeader("Location", "http://192.168.4.1", true); server.send(302); } });
  server.begin();
  digitalWrite(STBY, HIGH);
}

void loop() {
  dnsServer.processNextRequest();
  server.handleClient();
  if (motorStopTimer > 0 && millis() > motorStopTimer) { motorWifi(0); motorStopTimer = 0; }
  if (isAutoSleeping && millis() > modeRestoreTimer) { randomMode = backupMode; isAutoSleeping = false; }

  static unsigned long touchStartT = 0, touchReleaseT = 0;
  static int tCount = 0; static bool tActive = false, lpHandled = false;
  bool currTouch = isTouching();
  if (currTouch) { if (!tActive) { tActive = true; touchStartT = millis(); lpHandled = false; }
    else if (!lpHandled && (millis() - touchStartT > 800)) { handleLongPress(); lpHandled = true; tCount = 0; }
  } else { if (tActive) { tActive = false; if (!lpHandled) { tCount++; touchReleaseT = millis(); } } }
  if (tCount > 0 && !currTouch && (millis() - touchReleaseT > 1000)) {
    if (tCount == 1) handleSingleClick(); else if (tCount == 2) handleDoubleClick(); else if (tCount >= 3) handleTripleClick();
    tCount = 0;
  }

  if (currentState == STATE_COUNTDOWN) {
      unsigned long now = millis();
      unsigned long elapsed = now - countdownStartTime;
      long remainingMs = (countdownDurationMs > elapsed) ? (countdownDurationMs - elapsed) : 0;
      long remSec = remainingMs / 1000;
      if (remSec == 60 && !hasDone60sReminder) { hasDone60sReminder = true; playStrongReminder(); lastUpdateTick = millis(); }
      if (remainingMs == 0) { currentState = STATE_NORMAL; randomMode = backupMode; feedbackText = "结束"; feedbackStartTime = millis(); showingFaceInCountdown = false; motorWifi(0); }
      else {
          bool inBlockedRange = (remSec <= 55 && remSec >= 10);
          if (inBlockedRange) showingFaceInCountdown = false;
          else {
              faceIntervalAccumulator += (now - lastUpdateTick);
              if (faceIntervalAccumulator >= 40000) { showingFaceInCountdown = true; lastFaceShowTime = now; faceIntervalAccumulator = 0; }
              if (showingFaceInCountdown && (now - lastFaceShowTime >= 5000)) showingFaceInCountdown = false;
          }
      }
      lastUpdateTick = now;
  }

  if (displayFeedbackActive) {
    if (millis() - feedbackStartTime < feedbackDuration) {
      display.clearDisplay(); u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312);
      int16_t w = u8g2Fonts.getUTF8Width(feedbackText.c_str());
      u8g2Fonts.setCursor((128 - w) / 2, 40); u8g2Fonts.print(feedbackText); display.display();
    } else displayFeedbackActive = false;
  }
  else if (currentState != STATE_NORMAL) {
    display.clearDisplay();
    if (currentState == STATE_MENU_TIME) {
      u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312); display.drawRect(0, 22, 128, 20, SSD1306_WHITE);
      if (menuIndex > 0) { u8g2Fonts.setCursor(35, 16); u8g2Fonts.print(String(presetTimes[menuIndex - 1]) + " 分钟"); }
      u8g2Fonts.setCursor(35, 38); u8g2Fonts.print(String(presetTimes[menuIndex]) + " 分钟");
      if (menuIndex < 10) { u8g2Fonts.setCursor(35, 60); u8g2Fonts.print(String(presetTimes[menuIndex + 1]) + " 分钟"); }
    }
    else if (currentState == STATE_MENU_CONFIRM) {
      u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312); u8g2Fonts.setCursor(35, 25); u8g2Fonts.print(String(currentSelectedMinutes) + " 分钟");
      if (confirmMenuSelection == 0) display.fillRect(15, 40, 40, 20, SSD1306_WHITE); else display.drawRect(15, 40, 40, 20, SSD1306_WHITE);  
      if (confirmMenuSelection == 1) display.fillRect(75, 40, 40, 20, SSD1306_WHITE); else display.drawRect(75, 40, 40, 20, SSD1306_WHITE);  
      display.setTextColor(confirmMenuSelection == 0 ? SSD1306_BLACK : SSD1306_WHITE); u8g2Fonts.setCursor(19, 56); u8g2Fonts.print("开始");  
      display.setTextColor(confirmMenuSelection == 1 ? SSD1306_BLACK : SSD1306_WHITE); u8g2Fonts.setCursor(75, 56); u8g2Fonts.print("自定义");  
      display.setTextColor(SSD1306_WHITE);   
    }  
    else if (currentState == STATE_MENU_CUSTOM) {  
      u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312); u8g2Fonts.setCursor(10, 20); u8g2Fonts.print("自定义时间:");  
      u8g2Fonts.setCursor(45, 45); u8g2Fonts.print(String(currentSelectedMinutes) + " 分钟");  
    }  
    else if (currentState == STATE_COUNTDOWN) {  
        if (showingFaceInCountdown) roboEyes.update();
        else {
            unsigned long elapsed = millis() - countdownStartTime;
            long remMs = (countdownDurationMs > elapsed) ? (countdownDurationMs - elapsed) : 0;
            long remSec = remMs / 1000;
            // 30秒闪烁逻辑
            if (remSec >= 30 || (millis() % 500 < 250)) {
                int rMin = remSec / 60; int rSec = remSec % 60;  
                display.setTextSize(3); display.setCursor(22, 22);  
                if(rMin<10) display.print("0"); display.print(rMin); display.print(":");  
                if(rSec<10) display.print("0"); display.print(rSec);  
            }
            drawProgressBarBorder((float)remMs / countdownDurationMs);
        }
    }  
    if (currentState != STATE_COUNTDOWN || !showingFaceInCountdown) display.display();
  }
  else if (customTextActive) {
    if (!customTextLocked && (millis() - customTextStartTime >= customTextDuration)) { customTextActive = false; }
    else {
      display.clearDisplay(); u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312);
      if (textDisplayMode == 1) { int w = u8g2Fonts.getUTF8Width(textLine1.c_str()); u8g2Fonts.setCursor((128 - w) / 2, 40); u8g2Fonts.print(textLine1); }
      else if (textDisplayMode == 2) {
        int w1 = u8g2Fonts.getUTF8Width(textLine1.c_str()); int w2 = u8g2Fonts.getUTF8Width(textLine2.c_str());
        u8g2Fonts.setCursor((128 - w1) / 2, 28); u8g2Fonts.print(textLine1); u8g2Fonts.setCursor((128 - w2) / 2, 52); u8g2Fonts.print(textLine2);
      }
      else if (textDisplayMode == 3) {
        u8g2Fonts.setCursor(scrollX, 40); u8g2Fonts.print(customTextContent);
        static unsigned long lastSc = 0; if (millis() - lastSc > 25) { lastSc = millis(); scrollX -= 2; if (scrollX < -u8g2Fonts.getUTF8Width(customTextContent.c_str())) scrollX = 128; }
      }
      if (!customTextLocked) { int barW = map(millis() - customTextStartTime, 0, customTextDuration, 128, 0); display.drawFastHLine(0, 60, barW, SSD1306_WHITE); }
      display.display();
    }
  }
  else roboEyes.update();

  if (currentState == STATE_NORMAL && !isEyeLocked && !customTextLocked && (millis() - lastMoodTime >= moodInterval)) {
    lastMoodTime = millis(); currentMoodIdx = (currentMoodIdx + 1) % 4; roboEyes.setMood(moods[currentMoodIdx]);
  }
  static unsigned long lastMoveTick = 0;
  if (millis() - lastMoveTick > 40) {
    lastMoveTick = millis();
    if (!isAutoSleeping && currentState == STATE_NORMAL) {
      if (randomMode == RANDOM_SOFT && random(120) == 1) MOTOR(random(9), 25, 60, 1);
      else if (randomMode == RANDOM_NORMAL && random(100) == 1) MOTOR(random(9), 15, 50, random(15));
    }
  }
}
