}

  preferences.putString("w_fcast7", forecast7);

void updateWeatherTask() {
  if (WiFi.status() != WL_CONNECTED) return;
  HTTPClient http;
  String url = "http://api.seniverse.com/v3/weather/daily.json?key=" + seniverseKey + "&location=ip&language=zh-Hans&unit=c&start=0&days=7";
  http.begin(url);
  int httpCode = http.GET();
  if (httpCode == HTTP_CODE_OK) {
    String payload = http.getString();
    DynamicJsonDocument doc(4096); 
    DeserializationError error = deserializeJson(doc, payload);
    if (!error && doc["results"].size() > 0) {
      JsonObject loc = doc["results"][0]["location"];
      weatherData.city = loc["name"].as<String>();
      
      JsonArray daily = doc["results"][0]["daily"];
      if (daily.size() > 0) {
          weatherData.todayHighTemp = daily[0]["high"].as<String>();
          weatherData.todayTextDay = daily[0]["text_day"].as<String>();

          String fcastNormal = "";
          String fcast7Days = "";

          for (size_t i = 0; i < daily.size(); i++) {
              String dayName;
              if (i == 0) dayName = "今日";
              else if (i == 1) dayName = "明日";
              else if (i == 2) dayName = "后天";
              else dayName = daily[i]["date"].as<String>().substring(5); 

              String singleDay = dayName + ":" + daily[i]["text_day"].as<String>() + " " + daily[i]["low"].as<String>() + "-" + daily[i]["high"].as<String>() + "C";
              
              if (i < 3) {
                  fcastNormal += singleDay + "   ";
              }
              fcast7Days += singleDay + "   ";
          }
          
          weatherData.forecastData = fcastNormal;
          weatherData.forecastData7Days = fcast7Days;
          
          saveWeatherToCache(weatherData.city, daily[0]["date"].as<String>(), weatherData.todayHighTemp, weatherData.todayTextDay, fcastNormal, fcast7Days);
      }
    }
  }
  http.end();
}

String getCurrentTimeStr() {
  struct tm timeinfo;
  if (!getLocalTime(&timeinfo, 10)) return "--:--";
  char timeStringBuff[10];
  strftime(timeStringBuff, sizeof(timeStringBuff), "%H:%M", &timeinfo);
  return String(timeStringBuff);
}

bool isTouching() {
  pinMode(TOUCH_PIN, OUTPUT);
  digitalWrite(TOUCH_PIN, LOW);
  delayMicroseconds(1000);
  pinMode(TOUCH_PIN, INPUT_PULLUP);
  uint32_t count = 0;
  while (digitalRead(TOUCH_PIN) == LOW && count < 2000) { count++; asm volatile ("nop"); }
  return (count > 2);
}

void drawProgressBarBorder(float progress) {
  int remainingPixels = 384 * progress;
  int start = 384 - remainingPixels;
  int end = 384;
  if (start >= end) return;
  if (start < 128) { int s = start; int e = min(end, 128); display.drawFastHLine(s, 0, e - s, SSD1306_WHITE); }
  if (end > 128 && start < 192) { int s = max(start, 128) - 128; int e = min(end, 192) - 128; display.drawFastVLine(127, s, e - s, SSD1306_WHITE); }
  if (end > 192 && start < 320) { int s = max(start, 192) - 192; int e = min(end, 320) - 192; display.drawFastHLine(127 - e + 1, 63, e - s, SSD1306_WHITE); }
  if (end > 320) { int s = max(start, 320) - 320; int e = min(end, 384) - 320; display.drawFastVLine(0, 63 - e + 1, e - s, SSD1306_WHITE); }
}

void drawCountdownUI() {
    unsigned long elapsed = millis() - countdownStartTime;
    long remMs = (countdownDurationMs > elapsed) ? (countdownDurationMs - elapsed) : 0;
    long remSec = remMs / 1000;
    display.clearDisplay();
    display.setTextColor(SSD1306_WHITE); 
    
    int rHour = remSec / 3600;
    int rMin = (remSec % 3600) / 60; 
    int rSec = remSec % 60;  
    int halfMin = (countdownDurationMs / 60000) / 2;
    
    bool isBlinkNormal = false; bool isBlinkFast = false;
    if (remSec <= 15 && remSec >= 4) { isBlinkFast = true; }
    if (rSec >= 1 && rSec <= 4) {
        if (rMin == halfMin || rMin == 5 || rMin == 3 || rMin == 2) isBlinkNormal = true;
    }
    
    bool showText = true;
    if (isBlinkFast) showText = (millis() % 250 < 125);
    else if (isBlinkNormal) showText = (millis() % 500 < 250);
    
    if (showText) {
        if (remSec >= 3600) {
            // 当大于等于60分钟时，显示 H:MM:SS，去掉小时位的0保证最大字体刚好适配 128px 宽屏幕
            display.setTextSize(3); 
            display.setCursor(2, 22); // 精确偏移居中
            display.print(rHour); display.print(":");
            if(rMin < 10) display.print("0"); display.print(rMin); display.print(":");
            if(rSec < 10) display.print("0"); display.print(rSec);
        } else {
            // 小于60分钟，保持原来的大字体 MM:SS
            display.setTextSize(3); 
            display.setCursor(20, 22);
            if(rMin < 10) display.print("0"); display.print(rMin); display.print(":");  
            if(rSec < 10) display.print("0"); display.print(rSec);  
        }
    }
    
    drawProgressBarBorder((float)remMs / countdownDurationMs);
    display.display();
}

void smartDelay(unsigned long ms) {
    unsigned long start = millis();
    while (millis() - start < ms) {
        dnsServer.processNextRequest();
        server.handleClient();
        if (currentState == STATE_COUNTDOWN) drawCountdownUI();
        yield();
    }
}

    case 4: analogWrite(LF, speedL); analogWrite(LB, 0); analogWrite(RF, speedR); analogWrite(RB, 0); break; 
  }
}

/* ================= 触控动作逻辑 ================= */
void triggerModeSwitch() {
  if (backupMode == RANDOM_OFF) backupMode = RANDOM_SOFT;
  else if (backupMode == RANDOM_SOFT) backupMode = RANDOM_NORMAL;
  else backupMode = RANDOM_OFF;
  if(currentState != STATE_COUNTDOWN) randomMode = backupMode;
  feedbackText = (backupMode == RANDOM_OFF) ? "睡眠模式" : (backupMode == RANDOM_SOFT ? "摆动模式" : "好奇模式");
  displayFeedbackActive = true; feedbackStartTime = millis();
}

void triggerLockExpression() {
  if (customTextActive) {
    customTextLocked = !customTextLocked;
    feedbackText = customTextLocked ? "文字已锁定" : "文字已解锁";
  } else {
    isEyeLocked = !isEyeLocked;
    feedbackText = isEyeLocked ? "状态已锁定" : "状态已解锁";
  }
  displayFeedbackActive = true; feedbackStartTime = millis();
}

void handleSingleClick() {
  if (customTextLocked) return; 
  interruptWeatherIfUnlocked(); // 打断天气显示
  
  if (currentState == STATE_NORMAL) {
    if (customTextActive) { customTextActive = false; customTextLocked = false; }
    isEyeLocked = false; currentMoodIdx = (currentMoodIdx + 1) % 4;
    roboEyes.setMood(moods[currentMoodIdx]); lastMoodTime = millis();
  }
  else if (currentState == STATE_MENU_TIME) { menuIndex = (menuIndex + 1) % 13; }
  else if (currentState == STATE_MENU_CONFIRM) { confirmMenuSelection = (confirmMenuSelection == 0) ? 1 : 0; }
  else if (currentState == STATE_MENU_CUSTOM) {
    currentSelectedMinutes++;
    if (currentSelectedMinutes >= customMaxMinutes) currentSelectedMinutes = customBaseMinutes;
  }
}

void handleDoubleClick() {
  if (currentState == STATE_NORMAL || currentState == STATE_COUNTDOWN) { triggerModeSwitch(); }
  else if (currentState == STATE_MENU_TIME) { menuIndex = (menuIndex - 1 + 13) % 13; }
  else if (currentState == STATE_MENU_CUSTOM) {
    currentSelectedMinutes--;
    if (currentSelectedMinutes < customBaseMinutes) currentSelectedMinutes = customMaxMinutes - 1;
  }
}

void handleTripleClick() {
  if (currentState == STATE_NORMAL) { triggerLockExpression(); }
  else if (currentState == STATE_MENU_TIME) { currentState = STATE_NORMAL; }
  else if (currentState == STATE_MENU_CONFIRM) { currentState = STATE_MENU_TIME; }
  else if (currentState == STATE_MENU_CUSTOM) { currentState = STATE_MENU_CONFIRM; }
  else if (currentState == STATE_TEXT_LOCKED_HINT) { triggerLockExpression(); currentState = STATE_NORMAL; }
}

void handleLongPress() {
  if (customTextLocked) {
    if (millis() - firstLongPressTime < 1500) {
      customTextLocked = false; customTextActive = false; 
      currentState = STATE_MENU_TIME; menuIndex = 3; 
      firstLongPressTime = 0;
    } else {
      firstLongPressTime = millis();
      currentState = STATE_TEXT_LOCKED_HINT; hintStartTime = millis();
    }
    return;
  }
  
  interruptWeatherIfUnlocked(); // 打断天气显示
  
  if (currentState == STATE_NORMAL) { currentState = STATE_MENU_TIME; menuIndex = 3; }
  else if (currentState == STATE_MENU_TIME) {
    currentSelectedMinutes = presetTimes[menuIndex];
    if (currentSelectedMinutes >= 5) { currentState = STATE_MENU_CONFIRM; confirmMenuSelection = 0; }
    else { startCountdown(currentSelectedMinutes); }
  }
  else if (currentState == STATE_MENU_CONFIRM) {
    if (confirmMenuSelection == 0) { startCountdown(currentSelectedMinutes); }
    else {
      currentState = STATE_MENU_CUSTOM; customBaseMinutes = currentSelectedMinutes;
      customMaxMinutes = currentSelectedMinutes + 15;
      for(int i=0; i<12; i++) { if(presetTimes[i] == currentSelectedMinutes) { customMaxMinutes = presetTimes[i+1]; break; } }
    }
  }
  else if (currentState == STATE_MENU_CUSTOM) { startCountdown(currentSelectedMinutes); }
  else if (currentState == STATE_COUNTDOWN) {
    currentState = STATE_NORMAL; randomMode = backupMode;
    displayFeedbackActive = true; feedbackText = "倒计时解除"; feedbackStartTime = millis();
  }
}

void analyzeCustomText() {
  int totalChars = 0; int i = 0; int len = customTextContent.length();
  int totalWidth = u8g2Fonts.getUTF8Width(customTextContent.c_str());
  while (i < len) {
    char c = customTextContent[i];
    if ((c & 0x80) == 0) i += 1; else if ((c & 0xE0) == 0xC0) i += 2; else if ((c & 0xF0) == 0xE0) i += 3; else if ((c & 0xF8) == 0xF0) i += 4;
    totalChars++;
  }
  if (totalWidth <= 112 && totalChars <= 7) { textDisplayMode = 1; textLine1 = customTextContent; }
  else {
    textLine1 = ""; textLine2 = ""; int w1 = 0; i = 0; int charCount1 = 0;
    while (i < len) {
      char c = customTextContent[i]; int charLen = 1;
      if ((c & 0x80) == 0) charLen = 1; else if ((c & 0xE0) == 0xC0) charLen = 2; else if ((c & 0xF0) == 0xE0) charLen = 3; else if ((c & 0xF8) == 0xF0) charLen = 4;
      String singleChar = customTextContent.substring(i, i + charLen);
      int cw = u8g2Fonts.getUTF8Width(singleChar.c_str());
      if (charCount1 < 7 && w1 + cw <= 112) { textLine1 += singleChar; w1 += cw; charCount1++; }
      else { textLine2 += customTextContent.substring(i); break; }
      i += charLen;
    }
    textDisplayMode = 2;
  }
}

void initWiFi() {
  preferences.begin("wifi-cfg", false);
  sta_ssid = preferences.getString("ssid", ""); sta_password = preferences.getString("password", ""); ap_ssid = preferences.getString("ap_ssid", "桌面机器人");
  
  speedL = preferences.getInt("speedL", 255);
  speedR = preferences.getInt("speedR", 255);

  if (preferences.getBool("default_ap", false) || sta_ssid == "") { startAP(); }
  else {
    WiFi.mode(WIFI_STA); WiFi.begin(sta_ssid.c_str(), sta_password.c_str());
    unsigned long startTime = millis(); bool connected = false;
    while (millis() - startTime < 5000) {
      if (WiFi.status() == WL_CONNECTED) { connected = true; break; }
      display.clearDisplay(); u8g2Fonts.setFont(u8g2_font_wqy12_t_gb2312);
      u8g2Fonts.setCursor(0, 15); u8g2Fonts.print("WiFi模式：正在连接");
      u8g2Fonts.setCursor(0, 31); u8g2Fonts.print(sta_ssid);
      u8g2Fonts.setCursor(0, 47); u8g2Fonts.print("ip地址：" + WiFi.localIP().toString());
      u8g2Fonts.setCursor(0, 62); u8g2Fonts.print("信号强度：" + String(getSignalScore(WiFi.RSSI())) + "dBm");
      display.display(); delay(200);
    }
    if (connected) {
      isAPMode = false;
      configTime(8 * 3600, 0, "ntp.aliyun.com", "time.pool.aliyun.com");
      updateWeatherTask(); // 获取初次天气
      lastWeatherUpdate = millis();

      for (int i = 5; i > 0; i--) {
        display.clearDisplay(); u8g2Fonts.setFont(u8g2_font_wqy12_t_gb2312);
        u8g2Fonts.setCursor(0, 15); u8g2Fonts.print("WiFi模式：已连接");
        u8g2Fonts.setCursor(0, 31); u8g2Fonts.print(sta_ssid);
        u8g2Fonts.setCursor(0, 47); u8g2Fonts.print("ip地址：" + WiFi.localIP().toString());
        u8g2Fonts.setCursor(0, 62); u8g2Fonts.print("信号强度：" + String(getSignalScore(WiFi.RSSI())) + "dBm，" + String(i) + "s");
        display.display(); delay(1000);
      }
    } else { preferences.putBool("default_ap", true); startAP(); }
  }
  display.clearDisplay(); display.display();
}

void MOTOR(byte c, int t1, int t2, int Time) {
  if (randomMode == RANDOM_OFF || currentState == STATE_COUNTDOWN) return;
  for (int i = 0; i < Time; i++) {
    switch (c) {
      case 0: analogWrite(LF, 0); analogWrite(LB, 0); analogWrite(RF, 0); analogWrite(RB, 0); break;
      case 1: analogWrite(LF, 0); analogWrite(LB, speedL); analogWrite(RF, 0); analogWrite(RB, speedR); break;
      case 2: analogWrite(LF, speedL); analogWrite(LB, 0); analogWrite(RF, speedR); analogWrite(RB, 0); break;
      case 3: analogWrite(LF, 0); analogWrite(LB, speedL); analogWrite(RF, speedR); analogWrite(RB, 0); break;
      case 4: analogWrite(LF, speedL); analogWrite(LB, 0); analogWrite(RF, 0); analogWrite(RB, speedR); break;
      case 5: analogWrite(LF, 0); analogWrite(LB, speedL); analogWrite(RF, 0); analogWrite(RB, 0); break;
      case 6: analogWrite(LF, 0); analogWrite(LB, 0); analogWrite(RF, 0); analogWrite(RB, speedR); break;
      case 7: analogWrite(LF, speedL); analogWrite(LB, 0); analogWrite(RF, 0); analogWrite(RB, 0); break;
      case 8: analogWrite(LF, 0); analogWrite(LB, 0); analogWrite(RF, speedR); analogWrite(RB, 0); break;
    }
    smartDelay(t1); analogWrite(LF, 0); analogWrite(LB, 0); analogWrite(RF, 0); analogWrite(RB, 0); smartDelay(t2);
  }
}

void loop() {
  dnsServer.processNextRequest();
  server.handleClient();
  if (motorStopTimer > 0 && millis() > motorStopTimer) { motorWifi(0); motorStopTimer = 0; }
  if (isAutoSleeping && millis() > modeRestoreTimer) { randomMode = backupMode; isAutoSleeping = false; }

  // ---------------- 每小时更新一次天气数据 ----------------
  if (millis() - lastWeatherUpdate > 3600000 && WiFi.status() == WL_CONNECTED) {
    updateWeatherTask();
    lastWeatherUpdate = millis();
  }

  // ---------------- 自动显示天气与时间逻辑 ----------------
  if (millis() - bootTime > 180000) {
    struct tm timeinfo;
    if (getLocalTime(&timeinfo, 0)) { 
      int currentMin = timeinfo.tm_min;
      int currentSec = timeinfo.tm_sec;
      if ((currentMin % 5 == 0 || currentMin == 0 || currentMin == 30) && currentSec == 0) {
        if (currentMin != lastWeatherTriggerMinute) {
          bool eligible = false;
          // 防止倒计时菜单/进行中的倒计时被打断
          if (currentState == STATE_NORMAL && !customTextActive) eligible = true;
          if (customTextLocked) eligible = true;
          
          // 如果倒计时正在进行中，过滤掉快结束、最后五分钟、以及剩下一半时的时间
          if (currentState == STATE_COUNTDOWN) {
             unsigned long elapsed = millis() - countdownStartTime;
             long remSec = (countdownDurationMs > elapsed) ? (countdownDurationMs - elapsed) / 1000 : 0;
             long totalSec = countdownDurationMs / 1000;
             if (remSec <= 300 || remSec == totalSec / 2) {
                 eligible = false;
             }
          }

          if (eligible) {
            triggerWeatherDisplay();
            lastWeatherTriggerMinute = currentMin;
          }
        }
      }
    }
  }

  // ---------------- 触摸识别 ----------------
  static unsigned long touchStartT = 0, touchReleaseT = 0;
  static int tCount = 0; static bool tActive = false, lpHandled = false;
  const unsigned long T_LONG_PRESS = 600; const unsigned long T_CLICK_GAP = 1000; 
  bool currTouch = isTouching();
  if (currTouch) { if (!tActive) { tActive = true; touchStartT = millis(); lpHandled = false; } else if (!lpHandled && (millis() - touchStartT > T_LONG_PRESS)) { handleLongPress(); lpHandled = true; tCount = 0; } }
  else { if (tActive) { tActive = false; if (!lpHandled) { tCount++; touchReleaseT = millis(); } } }
  if (tCount > 0 && !currTouch && (millis() - touchReleaseT > T_CLICK_GAP)) { 
      if (tCount == 1) handleSingleClick(); 
      else if (tCount == 2) handleDoubleClick(); 
      else if (tCount == 3) {
          if (isWeatherTimeActive) {
              isEyeLocked = !isEyeLocked; 
              feedbackText = isEyeLocked ? "状态已锁定" : "状态已解锁";
              displayFeedbackActive = true; feedbackStartTime = millis();
          } else {
              handleTripleClick(); 
          }
      } 
      else if (tCount >= 4) { 
          if (currentState == STATE_NORMAL) {
              triggerWeatherDisplay();
          }
      }
      tCount = 0; 
  }

  // ---------------- 画面渲染逻辑 ----------------
  if (isWeatherTimeActive) {
      if (!isEyeLocked && millis() - weatherDisplayStartTime >= 8000) {
          isWeatherTimeActive = false;
      } else {
          display.clearDisplay();
          u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312);

          bool isOffline = (isAPMode || WiFi.status() != WL_CONNECTED);

          // 第一行：时间或离线标识
          String l1 = isOffline ? "离线天气" : getCurrentTimeStr();
          int tw = u8g2Fonts.getUTF8Width(l1.c_str());
          u8g2Fonts.setCursor((128 - tw) / 2, 16);
          u8g2Fonts.print(l1);

          String c_city = isOffline ? preferences.getString("w_city", "") : weatherData.city;
          if (c_city != "") {
             // 第二行：城市 日期/气温 天气 
             String l2;
             if (isOffline) {
                String c_date = preferences.getString("w_date", "");
                if(c_date.length() >= 10) c_date = c_date.substring(5); // 提取 MM-DD
                l2 = c_city + " " + c_date + " " + preferences.getString("w_text", "无");
             } else {
                l2 = c_city + " " + weatherData.todayHighTemp + "C " + weatherData.todayTextDay;
             }
             int lw = u8g2Fonts.getUTF8Width(l2.c_str());
             u8g2Fonts.setCursor((128 - lw) / 2, 38);
             u8g2Fonts.print(l2);
             
             // 第三行：预报滚动
             String fcast;
             if (isOffline) {
                 fcast = preferences.getString(isEyeLocked ? "w_fcast7" : "w_fcast", "未缓存天气预报");
             } else {
                 fcast = isEyeLocked ? weatherData.forecastData7Days : weatherData.forecastData;
             }
             
             u8g2Fonts.setCursor(weatherScrollX, 60);
             u8g2Fonts.print(fcast);

             static unsigned long lastSc = 0;
             if (millis() - lastSc > 25) {
                 lastSc = millis(); weatherScrollX -= 2;
                 if (weatherScrollX < -u8g2Fonts.getUTF8Width(fcast.c_str())) {
                     weatherScrollX = 128;
                 }
             }
          } else {
             u8g2Fonts.setCursor(10, 45); u8g2Fonts.print("获取天气中...");
          }
          display.display();
      }
  }
  else if (currentState == STATE_COUNTDOWN) {
      unsigned long now = millis();
      unsigned long elapsed = now - countdownStartTime;
      long remMs = (countdownDurationMs > elapsed) ? (countdownDurationMs - elapsed) : 0;
      long remSec = remMs / 1000;
      long totalSec = countdownDurationMs / 1000;
      
      if (remSec == 10 && !hasDone8sReminder) { hasDone8sReminder = true; playStrongReminder(); playStrongReminder(); }
      if (remSec == totalSec / 2 && !hasDoneHalfReminder && totalSec > 0) { hasDoneHalfReminder = true; playStrongReminder(); }
      
      if (remMs == 0) { 
          currentState = STATE_NORMAL; 
          randomMode = backupMode; 
          feedbackText = "结束"; 
          feedbackStartTime = millis(); 
          playStrongReminder(); 
      } else {
          drawCountdownUI();
      }
  }
  else if (displayFeedbackActive) {
    if (millis() - feedbackStartTime < feedbackDuration) {
      display.clearDisplay(); u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312);
      int16_t w = u8g2Fonts.getUTF8Width(feedbackText.c_str());
      u8g2Fonts.setCursor((128 - w) / 2, 40); u8g2Fonts.print(feedbackText); display.display();
    } else displayFeedbackActive = false;
  }
  else if (currentState == STATE_TEXT_LOCKED_HINT) {
    display.clearDisplay(); u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312);
    u8g2Fonts.setCursor(10, 25); u8g2Fonts.print("已锁定文字"); u8g2Fonts.setCursor(10, 50); u8g2Fonts.print("再次长按解锁");
    display.display(); if (millis() - hintStartTime > 3000) currentState = STATE_NORMAL;
  }
  else if (currentState != STATE_NORMAL) {
    display.clearDisplay();
    if (currentState == STATE_MENU_TIME) {
      u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312); display.drawRect(0, 22, 128, 20, SSD1306_WHITE);
      if (menuIndex > 0) { u8g2Fonts.setCursor(35, 16); u8g2Fonts.print(String(presetTimes[menuIndex - 1]) + " 分钟"); }
      u8g2Fonts.setCursor(35, 38); u8g2Fonts.print(String(presetTimes[menuIndex]) + " 分钟");
      if (menuIndex < 12) { u8g2Fonts.setCursor(35, 60); u8g2Fonts.print(String(presetTimes[menuIndex + 1]) + " 分钟"); }
    }
    else if (currentState == STATE_MENU_CONFIRM) {
      u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312); 
      u8g2Fonts.setCursor(35, 25); 
      u8g2Fonts.print(String(currentSelectedMinutes) + " 分钟");
      
      // 文字纯净显示，不再填充反色色块，也不会有竖线
      u8g2Fonts.setCursor(20, 55); 
      u8g2Fonts.print("开始");  
      u8g2Fonts.setCursor(72, 55); 
      u8g2Fonts.print("自定义");  

      // 只有一个空心的选择框，框住当前选项
      if (confirmMenuSelection == 0) {
          display.drawRect(16, 39, 40, 20, SSD1306_WHITE); // 框选“开始”
      } else {
          display.drawRect(68, 39, 56, 20, SSD1306_WHITE); // 框选“自定义”
      }
    }  
    else if (currentState == STATE_MENU_CUSTOM) {  
      u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312); u8g2Fonts.setCursor(10, 20); u8g2Fonts.print("自定义时间:");  
      u8g2Fonts.setCursor(45, 45); u8g2Fonts.print(String(currentSelectedMinutes) + " 分钟");  
    }  
    display.display();
  }
  else if (customTextActive) {
    if (!customTextLocked && (millis() - customTextStartTime >= customTextDurationLimit)) { customTextActive = false; }
    else {
      display.clearDisplay(); u8g2Fonts.setFont(u8g2_font_wqy16_t_gb2312);
      if (textDisplayMode == 1) { int w = u8g2Fonts.getUTF8Width(textLine1.c_str()); u8g2Fonts.setCursor((128 - w) / 2, 40); u8g2Fonts.print(textLine1); }
      else if (textDisplayMode == 2) {
        int w1 = u8g2Fonts.getUTF8Width(textLine1.c_str()); int w2 = u8g2Fonts.getUTF8Width(textLine2.c_str());
        u8g2Fonts.setCursor((128 - w1) / 2, 28); u8g2Fonts.print(textLine1); u8g2Fonts.setCursor((128 - w2) / 2, 52); u8g2Fonts.print(textLine2);
      }
      else if (textDisplayMode == 3) {
        u8g2Fonts.setCursor(customTextScrollX, 40);
        u8g2Fonts.print(customTextContent);
        static unsigned long lastSc = 0;
        if (millis() - lastSc > 20) { 
          lastSc = millis();
          customTextScrollX -= 2;
          if (customTextScrollX < -u8g2Fonts.getUTF8Width(customTextContent.c_str())) {
            customTextScrollX = 128; 
          }
        }
      }
      display.display();
    }
  }
  else roboEyes.update(); 

  if (currentState == STATE_NORMAL && !isEyeLocked && !customTextLocked && !isWeatherTimeActive && (millis() - lastMoodTime >= moodInterval)) {
    lastMoodTime = millis(); currentMoodIdx = (currentMoodIdx + 1) % 4; roboEyes.setMood(moods[currentMoodIdx]);
  }
  static unsigned long lastMoveTick = 0;
  if (millis() - lastMoveTick > 40) {
    lastMoveTick = millis();
    if (!isAutoSleeping && currentState == STATE_NORMAL && !isWeatherTimeActive) {
      if (randomMode == RANDOM_SOFT && random(120) == 1) MOTOR(random(9), 25, 60, 1);
      else if (randomMode == RANDOM_NORMAL && random(100) == 1) MOTOR(random(9), 15, 50, random(15));
    }
  }
}
